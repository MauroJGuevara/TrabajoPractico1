// subproceso del punto 1 

SubProceso ventaPasaje(asientosTotales,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba,datoPasajeros,pasajero,vuelos)
	definir NombrePasajero, rutaPasajero, equipajePasajero, pasajeroFrecuente,telPasajero Como Caracter
	definir numeroAsiento, dniPasajero Como Entero
	definir montoVuelo Como Real
	
	rutaPasajero = verificacionRuta(vuelos,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba)
	datoPasajeros[pasajero,0] = rutaPasajero
	
	segun rutaPasajero
		"Buenos Aires - Bariloche":
			asientosBa_Bar = asientosBa_Bar - 1
		"Bueno Aires - Salta":
			asientosBa_Salta = asientosBa_Salta - 1
		"Rosario - Buenos Aires":
			asientosRos_Ba = asientosRos_Ba - 1
		"Mar Del Plata - Mendoza":
			asientosMdp_Men = asientosMdp_Men - 1
	FinSegun
	
	escribir "Ingrese el nombre del pasajero: "
	leer NombrePasajero
	datoPasajeros[pasajero,1] = NombrePasajero
	
	dniPasajero = ingresoDni
	datoPasajeros[pasajero,2] = ConvertirATexto(dniPasajero)
	
	Mientras Longitud(telPasajero) < 12 o Longitud(telPasajero) > 13
		escribir "Ingrese su numero de telefono: "
		leer telPasajero
		si Longitud(telPasajero) < 12 o Longitud(telPasajero) > 13
			escribir "numero invalido"
		FinSi
	FinMientras
	
	datoPasajeros[pasajero,3] = telPasajero
	
	equipajePasajero = verifacionEquipaje
	datoPasajeros[pasajero,4] = equipajePasajero
	
	Mientras Longitud(pasajeroFrecuente) <> 4
		Escribir "Ingrese los 4 digitos de su numero de pasajero frecuente: "
		leer pasajeroFrecuente
		si Longitud(pasajeroFrecuente) <> 4
			escribir "numero invalido"
		FinSi
	FinMientras
	datoPasajeros[pasajero,5] = pasajeroFrecuente
	
	numeroAsiento = asignaAsiento(rutaPasajero,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba)
	datoPasajeros[pasajero,6] = ConvertirATexto(numeroAsiento)
	
	montoVuelo = montoFinal(rutaPasajero,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba,equipajePasajero)
	datoPasajeros[pasajero,7] = ConvertirATexto(montoVuelo)
	
	para i = 0 hasta 7 Hacer
		escribir "   ", datoPasajeros[pasajero,i]
	FinPara
FinSubProceso

SubProceso return = verificacionRuta(vuelos,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba)
	Definir return Como Caracter
	definir bandera, opcionVuelo Como Entero
	
repetir
	escribir "Ingrese la opcion de la ruta que desea : "
	Para i = 0 hasta 3 Hacer
		escribir " - ", i+1 " - ",vuelos[i]
	FinPara
	Repetir
		leer opcionVuelo
		si opcionVuelo< 1 o opcionVuelo > 4
			escribir "opcion incorrecta"
			bandera = 0
		FinSi
	hasta que opcionVuelo < 4 y opcionVuelo > 0
	
	segun opcionVuelo
		1:
			si asientosBa_Bar = 0
				Escribir "no hay asientos disponibles, disculpe las molestias"
				bandera = 0
			sino
				return = vuelos[0]
				bandera = 1
			FinSi
		2:
			si asientosBa_Salta = 0
				Escribir "no hay asientos disponibles, disculpe las molestias"
				bandera = 0
			sino
				return = vuelos[1]
				bandera = 1
			FinSi
		3: 
			si asientosRos_Ba = 0
				Escribir "no hay asientos disponibles, disculpe las molestias"
				bandera = 0
			sino
				return = vuelos[2]
				bandera = 1
			FinSi
		4:
			si asientosMdp_Men = 0
				Escribir "no hay asientos disponibles, disculpe las molestias"
				bandera = 0
			sino
				return = vuelos[3]
				bandera = 1
			FinSi
	FinSegun
Mientras Que bandera = 0
escribir "Destino seleccionado: ", return
	
FinSubProceso

SubProceso return = verifacionEquipaje
	definir return, optEquipaje Como Caracter
	Repetir
		escribir "Desea despachar el equipaje en bodega? Si/No"
		Leer optEquipaje
		si Mayusculas(optEquipaje) <> "SI" y Mayusculas(optEquipaje) <> "NO"
			escribir "opcion incorrecta"
		FinSi
	hasta que Mayusculas(optEquipaje) = "SI" o Mayusculas(optEquipaje) = "NO"
	
	si Mayusculas(optEquipaje) = "SI"
		return = "Verdadero"
	sino
		return = "Falso"
	FinSi
FinSubProceso

SubProceso return = asignaAsiento(rutaPasajero,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba)
	definir return Como entero
	segun rutaPasajero
		"Buenos Aires - Bariloche":
			return = asientosBa_Bar + 1
		"Bueno Aires - Salta":
			return = asientosBa_Salta + 1
		"Rosario - Buenos Aires":
			return = asientosRos_Ba + 1
		"Mar Del Plata - Mendoza":
			return = asientosMdp_Men + 1
	FinSegun
	
FinSubProceso

//subproceso de monto final del vuelo 

SubProceso return = montoFinal(rutaPasajero,asientosBa_Bar,asientosBa_Salta,asientosMdp_Men,asientosRos_Ba,equipajePasajero)
	definir return, montoBase como real
	segun rutaPasajero
		"Buenos Aires - Bariloche":
			si asientosBa_Bar + 1 >= 100
				montoBase = 150000
			SiNo
				si asientosBa_Bar + 1 >= 60 
					montoBase = 150000 * 1.10
				sino
					montoBase = 180000
				FinSi
			FinSi
		"Bueno Aires - Salta":
			si asientosBa_Salta + 1 >= 100
				montoBase = 120000
			SiNo
				si asientosBa_Salta + 1 >= 60 
					montoBase = 120000 * 1.10
				sino
					montoBase = 150000
				FinSi
			FinSi
		"Rosario - Buenos Aires":
			si asientosRos_Ba + 1 >= 70
				montoBase = 70000
			SiNo
				si asientosRos_Ba + 1 >= 40 
					montoBase = 70000 * 1.15
				sino
					montoBase = 95000
				FinSi
			FinSi
		"Mar Del Plata - Mendoza":
			si asientosMdp_Men + 1 >= 70
				montoBase = 95000
			SiNo
				si asientosMdp_Men + 1 >= 40 
					montoBase = 95000 * 1.15
				sino
					montoBase = 125000
				FinSi
			FinSi
	FinSegun
	si equipajePasajero = "Verdadero"
		return = montoBase * 1.05
	SiNo
		return = montoBase
	FinSi
FinSubProceso
